function CanvasDragDrop(obj) { this.el = obj.el; this.ctx = obj.ctx; this.offset = obj.offset || { x: 0, y: 0 }; this.ddElMap = {}; this.ddElList = []; this.ddEl = {}; this.mousePos = {}; this.ddElTrack = undefined; this.scaleRatio = { x: 1, y: 1 } this.init(); } Object.defineProperties(CanvasDragDrop.prototype, { init: { value: init }, destory: { value: destory }, addDragDropElement: { value: addDragDropElement }, addTextElement: { value: addTextElement }, addImageElement: { value: addImageElement }, render: { value: render }, isSelectDDEl: { value: isSelectDDEl }, matchDDEl: { value: matchDDEl }, drawOutline: { value: drawOutline }, subscribe: { value: subscribe }, scaleDDEl: { value: scaleDDEl } }); function init() { if (!this.el.onmousedown) this.el.onmousedown = mouseDown.bind(this); if (!this.el.onmouseup) this.el.onmouseup = mouseUp.bind(this); if (!this.el.onmousemove) this.el.onmousemove = mouseMove.bind(this); const rect = this.el.getBoundingClientRect(); this.scaleRatio.x = this.el.width / rect.width; this.scaleRatio.y = this.el.height / rect.height; } function destory() { this.el.onmousedown = undefined; this.el.onmouseup = undefined; this.el.onmousemove = undefined; } function mouseDown(ev) { this.mousePos = this.scaleDDEl({ x: ev.offsetX, y: ev.offsetY }); let matchIndex = this.matchDDEl(this.mousePos), matchEl = this.ddElList[matchIndex]; if (matchIndex !== -1) { this.ddElList.splice(matchIndex, 1); this.ddElList.unshift(matchEl); this.ddEl = matchEl; } } function mouseMove(ev) { let mouseMovePos = this.scaleDDEl({ x: ev.offsetX, y: ev.offsetY }); if (this.isSelectDDEl()) { let offsetPos = { x: this.mousePos.x - mouseMovePos.x, y: this.mousePos.y - mouseMovePos.y }; this.ddEl.x -= offsetPos.x; this.ddEl.y -= offsetPos.y; this.render(); this.mousePos = mouseMovePos; if (this.ddElTrack) { this.ddElTrack.call(this, { x: this.ddEl.x - this.offset.x, y: this.ddEl.y - this.offset.y }); } } else { let matchIndex = this.matchDDEl(mouseMovePos), matchEl = this.ddElList[matchIndex]; if (matchIndex !== -1) { this.render(); this.drawOutline(matchEl); } } } function mouseUp(ev) { this.ddEl = {}; } function addDragDropElement(el, callback) { let key = `${el.x}-${el.y}-${el.width}-${el.height}`; el.id = key; el.render = callback; el.x += this.offset.x; el.y += this.offset.y; this.ddElMap[key] = el; this.ddElList.unshift(el); // 箭头函数无法绑定this，call，apply，bind的第一个参数会被忽略。 callback.call(this, this.ctx, el); } function addTextElement(el, callback) { let ctx = this.ctx; ctx.save(); el.type = 'text'; el.fontFamily = el.fontFamily || 'serif'; el.fontSize = el.fontSize || 14; el.fontColor = el.fontColor || 'rgb(0, 0, 0)'; el.font = `${el.fontSize}px ${el.fontFamily}`; el.textBaseline = 'top'; ctx.font = el.font; el.width = el.width || ctx.measureText(el.text).width; el.height = el.height || el.fontSize + 4; ctx.restore(); this.addDragDropElement(el, callback); } function addImageElement(el, callback) { let image = el.image; el.type = 'image'; el.width = el.width || image.width; el.height = el.height || image.height; this.addDragDropElement(el, callback); } function render() { this.ctx.clearRect(0, 0, this.ctx.canvas.width, this.ctx.canvas.height); for (let i = this.ddElList.length, el; i--;) { el = this.ddElList[i]; el.render.call(this, this.ctx, el); } } /** * 判断是否鼠标点击选中了某个拖放元素 * @return {*|boolean} */ function isSelectDDEl() { return this.ddEl !== undefined && this.ddEl.x !== undefined && this.ddEl.y !== undefined; } /** * 匹配拖放元素 * @param mousePos * @return {number} */ function matchDDEl(mousePos) { for (let i = 0, len = this.ddElList.length, el; i < len; i++) { el = this.ddElList[i]; if ((mousePos.x > el.x && mousePos.x < (el.x + el.width)) && (mousePos.y > el.y && mousePos.y < (el.y + el.height))) { return i; } } return -1; } function drawOutline(el) { let ctx = this.ctx; ctx.save(); ctx.strokeStyle = 'rgb(0, 0, 255)'; if (el.type === 'text') { ctx.strokeRect(el.x, el.y, el.width, el.height); } else { ctx.strokeRect(el.x, el.y, el.width, el.height); } ctx.restore(); } function subscribe(callback) { this.ddElTrack = callback; } /** * 缩放拖拽元素 */ function scaleDDEl(el) { if (el.x !== undefined) el.x *= this.scaleRatio.x; if (el.y !== undefined) el.y *= this.scaleRatio.y; if (el.width !== undefined) el.width *= this.scaleRatio.x; if (el.height !== undefined) el.height *= this.scaleRatio.y; return el; } export default CanvasDragDrop;
